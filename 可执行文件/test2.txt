void main()
{
	int fahr,celsius;
	int lower,upper,step;

	lower=0;
	upper=300;
	step=20;

	fahr=lower;
	printf("//while statement");
	while(fahr<=upper)
	{
		celsius=5*(fahr-32)/9;
		printf(fahr);
		printf(celsius);
		fahr=fahr+step;
	}
	printf("//while statement");
}
